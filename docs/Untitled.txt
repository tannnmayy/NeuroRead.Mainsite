  document.getElementById('simplifier-modal').addEventListener('click',function(e){if(e.target===this)closeSimplifier()});
  document.addEventListener('keydown',e=>{if(e.key==='Escape')closeSimplifier()});
  document.addEventListener('keydown',e=>{if((e.ctrlKey||e.metaKey)&&e.key==='Enter')runSimplifier()});

  // Dyslexia / Audio
  let dyslexiaOn=false, audioOn=false;
  window.toggleDyslexia=function(){

    audioOn=!audioOn;
    const bars=document.getElementById('wave-bars'), btn=document.getElementById('audio-btn');
    if(audioOn){bars.classList.add('playing');btn.classList.add('bg-moss','text-cream');

      if(txt&&txt!=='Simplified text will appear here once you run an analysis.'&&window.speechSynthesis){

    const btn=document.getElementById('simplify-btn');

  function applyMetrics(d){
    document.getElementById('score-val').textContent=d.originalScore;
    document.getElementById('score-bar').style.width=d.originalScore+'%';
    const diff=d.difficulty.toLowerCase();
    const bc={high:'bg-red-50 text-red-600 border-red-200',moderate:'bg-clay/10 text-clay border-clay/20',low:'bg-moss/10 text-moss border-moss/20'};
    document.getElementById('score-badge').textContent=diff.toUpperCase();
    document.getElementById('score-badge').className='text-xs font-medium px-3 py-1 rounded-full border '+(bc[diff]||bc.moderate);
    document.getElementById('reading-time-val').textContent=d.readingTime;
    document.getElementById('difficulty-val').textContent=d.difficulty;

    }
  }

  // Main API call
  window.runSimplifier=async function(){
    const text=document.getElementById('input-text').value.trim();
    if(!text){alert('Please enter some text to simplify.');return;}
    const profile=document.getElementById('profile-select').value;
    setLoading(true);
    try{
      const res=await fetch('https://api.anthropic.com/v1/messages',{
        method:'POST',
        headers:{'Content-Type':'application/json'},
        body:JSON.stringify({
          model:'claude-sonnet-4-20250514',
          max_tokens:1000,
          system:`You are a cognitive accessibility AI. Return ONLY valid JSON (no markdown, no extra text):
{"simplifiedText":"plain-language rewrite","originalScore":<1-100>,"readingTime":"<e.g. 3 min>","difficulty":"<High|Moderate|Low>","reduction":<% integer>,"intensity":<1-100>,"impactSummary":"2-3 sentence analysis","keywords":["up to 6 key terms"]}
Reading profile: ${profile}.`,
          messages:[{role:'user',content:text}]
        })
      });
      const data=await res.json();
      const raw=data.content.map(b=>b.text||'').join('');
      const parsed=JSON.parse(raw.replace(/```json|```/g,'').trim());
      document.getElementById('simplified-output').textContent=parsed.simplifiedText;
      applyMetrics(parsed);
      addLiveSession(parsed);
    }catch(err){
      document.getElementById('simplified-output').textContent='Error: Could not reach the API. Please try again.';
      console.error(err);
    }finally{setLoading(false);}
  };
